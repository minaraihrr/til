```python
# def f(n):
#   return n if n <= 1 else f(n - 1) + f(n - 2)

# メモ化再帰
memo = {}
def f(n):
  if n in memo:
    return memo[n]
  if n <= 1:
    ret = n
  else:
    ret = f(n - 1) + f(n - 2)
  memo[n] = ret
  return ret
print(f(70))

# DP
N = 70
dp = [[] for _ in range(N + 1)]
dp[0] = 0
dp[1] = 1
for n in range(2, N + 1):
  dp[n] = dp[n - 1] + dp[n - 2]
print(dp[N])
```
